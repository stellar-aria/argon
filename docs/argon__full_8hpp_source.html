211 template <typename CommutableOpType>
212 scalar_type Reduce(CommutableOpType op) {
213 auto rev = this->SwapDoublewords();
214 auto sum = op(*this, rev);
215 if constexpr (lanes == 16) {
216 sum = op(*this, sum.Reverse16bit());
217 }
218 if constexpr (lanes == 8 || lanes == 16) {
219 sum = op(*this, sum.Reverse32bit());
220 }
221 if constexpr (lanes == 4 || lanes == 8 || lanes == 16) {
222 sum = op(*this, sum.Reverse64bit());
223 }
224 return sum[0];
225 }
