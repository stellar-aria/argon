209 template <typename CommutableOpType>
210 scalar_type Reduce(CommutableOpType op) {
211 auto rev = this->SwapDoublewords();
212 auto sum = op(*this, rev);
213 if constexpr (lanes == 16) {
214 sum = op(*this, sum.Reverse16bit());
215 }
216 if constexpr (lanes == 8 || lanes == 16) {
217 sum = op(*this, sum.Reverse32bit());
218 }
219 if constexpr (lanes == 4 || lanes == 8 || lanes == 16) {
220 sum = op(*this, sum.Reverse64bit());
221 }
222 return sum[0];
223 }
