175template <typename CondType, typename ScalarType>
176 requires std::is_arithmetic_v<ScalarType> && std::is_same_v<CondType, typename Argon<ScalarType>::argon_result_type>
177class CondMonad : public std::pair<CondType, Argon<ScalarType>> {
178 using ArgonType = Argon<ScalarType>;
179
180 public:
181 using PairType = std::pair<CondType, ArgonType>;
182 using PairType::PairType;
183
184 constexpr CondType condition() const { return this->first; }
185 constexpr ArgonType value() const { return this->second; }
186
187 CondMonad else_if_(CondType new_condition, ArgonType new_value) const {
188 CondType exclusive_condition = new_condition & ~condition();
189 if constexpr (std::is_same_v<CondType, ArgonType>) {
190 // if condition and value are the same type, we can just use the value directly
191 new_value = (exclusive_condition & new_value) | (~exclusive_condition & value());
192 } else {
193 new_value = ((exclusive_condition & new_value.template As<typename CondType::scalar_type>()) |
194 (~exclusive_condition & value().template As<typename CondType::scalar_type>()))
195 .template As<typename ArgonType::scalar_type>();
196 }
197 return CondMonad{condition() | new_condition, new_value};
198 }
199
200 template <typename FunctionType>
201 CondMonad else_if_(CondType condition, FunctionType func) const {
202 return else_if_(condition, func());
203 }
204
205 CondMonad else_if_(CondType condition, ScalarType value) const { return else_if_(condition, ArgonType{value}); }
206
207 ArgonType else_(ArgonType new_value) const {
208 if constexpr (std::is_same_v<CondType, ArgonType>) {
209 // if condition and value are the same type, we can just use the value directly
210 return (condition() & new_value) | (~condition() & value());
211 } else {
212 // negate mask to get what's left, and select those lanes from the input
213 return ((~condition() & new_value.template As<typename CondType::scalar_type>()) |
214 (condition() & value().template As<typename CondType::scalar_type>()))
215 .template As<typename ArgonType::scalar_type>();
216 }
217 }
218
219 template <typename FunctionType>
220 ArgonType else_(FunctionType func) const {
221 return else_(func());
222 }
223
224 ArgonType else_(ScalarType value) const { return else_(ArgonType{value}); }
225};
