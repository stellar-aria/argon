107template <typename CondType, typename ScalarType>
108 requires std::is_arithmetic_v<ScalarType> && std::is_same_v<CondType, typename Argon<ScalarType>::argon_result_type>
109class CondMonad : public std::pair<CondType, Argon<ScalarType>> {
110 using ArgonType = Argon<ScalarType>;
111
112 public:
113 using PairType = std::pair<CondType, ArgonType>;
114 using PairType::PairType;
115
116 constexpr CondType condition() const { return this->first; }
117 constexpr ArgonType value() const { return this->second; }
118
119 CondMonad else_if_(CondType new_condition, ArgonType new_value) const {
120 CondType exclusive_condition = new_condition & ~condition();
121 if constexpr (std::is_same_v<CondType, ArgonType>) {
122 // if condition and value are the same type, we can just use the value directly
123 new_value = (exclusive_condition & new_value) | (~exclusive_condition & value());
124 } else {
125 new_value = ((exclusive_condition & new_value.template As<typename CondType::scalar_type>()) |
126 (~exclusive_condition & value().template As<typename CondType::scalar_type>()))
127 .template As<typename ArgonType::scalar_type>();
128 }
129 return CondMonad{condition() | new_condition, new_value};
130 }
131
132 template <typename FunctionType>
133 CondMonad else_if_(CondType condition, FunctionType func) const {
134 return else_if_(condition, func());
135 }
136
137 CondMonad else_if_(CondType condition, ScalarType value) const { return else_if_(condition, ArgonType{value}); }
138
139 ArgonType else_(ArgonType new_value) const {
140 if constexpr (std::is_same_v<CondType, ArgonType>) {
141 // if condition and value are the same type, we can just use the value directly
142 return (condition() & new_value) | (~condition() & value());
143 } else {
144 // negate mask to get what's left, and select those lanes from the input
145 return ((~condition() & new_value.template As<typename CondType::scalar_type>()) |
146 (condition() & value().template As<typename CondType::scalar_type>()))
147 .template As<typename ArgonType::scalar_type>();
148 }
149 }
150
151 template <typename FunctionType>
152 ArgonType else_(FunctionType func) const {
153 return else_(func());
154 }
155
156 ArgonType else_(ScalarType value) const { return else_(ArgonType{value}); }
157};
