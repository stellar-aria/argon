460 template <size_t n>
461 ace static std::array<argon_type, n> LoadMulti(const scalar_type* ptr) {
462 static_assert(n > 1 && n < 5, "LoadMulti can only be performed with a size of 2, 3, or 4");
463#if defined(__clang__) || (__GNUC__ > 13)
464 if constexpr (n == 2) {
465 return argon::to_array(simd::load1_x2(ptr).val);
466 } else if constexpr (n == 3) {
467 return argon::to_array(simd::load1_x3(ptr).val);
468 } else if constexpr (n == 4) {
469 return argon::to_array(simd::load1_x4(ptr).val);
470 }
471#else
472 if constexpr (n == 2) {
473 auto a = simd::load1(ptr);
474 auto b = simd::load1(ptr + lanes);
475 return {a, b};
476 } else if constexpr (n == 3) {
477 auto a = simd::load1(ptr);
478 auto b = simd::load1(ptr + lanes);
479 auto c = simd::load1(ptr + 2 * lanes);
480 return {a, b, c};
481 } else if constexpr (n == 4) {
482 auto a = simd::load1(ptr);
483 auto b = simd::load1(ptr + lanes);
484 auto c = simd::load1(ptr + 2 * lanes);
485 auto d = simd::load1(ptr + 3 * lanes);
486 return {a, b, c, d};
487 }
488#endif
